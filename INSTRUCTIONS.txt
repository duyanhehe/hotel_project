STEPS
1. Create venv and install everything in requirements.txt
2. Create your own dbfunc.py file like this (change hostname, username, password to your own)
3. make sure your sql server is running

import mysql.connector
from mysql.connector import errorcode

hostname    = "YourHostname"
db          = "hotel_db"
username    = "YourUsername"
passwd  = "YourPassword"
    
def getConnection():    
    try:
        conn = mysql.connector.connect(host=hostname,                              
                            user=username,
                            db = db,
                            password=passwd)  
    except mysql.connector.Error as err:
        if err.errno == errorcode.ER_ACCESS_DENIED_ERROR:
            print('User name or Password is not working')
        elif err.errno == errorcode.ER_BAD_DB_ERROR:
            print('Database does not exist')
        else:
            print(err)                        
    else:  
        return conn
